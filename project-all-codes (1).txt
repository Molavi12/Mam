/**
 * Baleh Bot API Client
 * Baleh messenger API is similar to Telegram Bot API
 * Base URL: https://tapi.bale.ai/bot{token}/
 */

const BALEH_API_BASE = 'https://tapi.bale.ai';

interface BalehResponse {
  ok: boolean;
  result?: any;
  description?: string;
  error_code?: number;
}

interface BalehUpdate {
  update_id: number;
  message?: BalehMessage;
  edited_message?: BalehMessage;
}

interface BalehMessage {
  message_id: number;
  from?: BalehUser;
  chat: BalehChat;
  date: number;
  text?: string;
  caption?: string;
  photo?: BalehPhotoSize[];
  video?: BalehVideo;
  document?: BalehDocument;
  reply_to_message?: BalehMessage;
}

interface BalehUser {
  id: number;
  is_bot: boolean;
  first_name: string;
  last_name?: string;
  username?: string;
}

interface BalehChat {
  id: number;
  type: string;
  title?: string;
  username?: string;
  first_name?: string;
}

interface BalehPhotoSize {
  file_id: string;
  file_unique_id: string;
  width: number;
  height: number;
  file_size?: number;
}

interface BalehVideo {
  file_id: string;
  file_unique_id: string;
  width: number;
  height: number;
  duration: number;
  file_name?: string;
  mime_type?: string;
  file_size?: number;
}

interface BalehDocument {
  file_id: string;
  file_unique_id: string;
  file_name?: string;
  mime_type?: string;
  file_size?: number;
}

interface BalehFile {
  file_id: string;
  file_unique_id: string;
  file_size?: number;
  file_path?: string;
}

export class BalehClient {
  private token: string;
  private lastUpdateId: number = 0;
  private polling: boolean = false;
  private pollTimeout: NodeJS.Timeout | null = null;
  private messageHandler: ((message: BalehMessage) => Promise<void>) | null = null;
  // Message deduplication: track recently processed message IDs
  private processedMessageIds: Set<number> = new Set();
  private readonly MAX_PROCESSED_IDS = 1000; // Keep last 1000 message IDs
  // Processing lock to prevent concurrent handler execution
  private processingInProgress: boolean = false;
  // Pending messages queue
  private messageQueue: BalehMessage[] = [];

  constructor(token: string) {
    this.token = token;
  }

  private get baseUrl(): string {
    return `${BALEH_API_BASE}/bot${this.token}`;
  }

  async apiCall(method: string, params: Record<string, any> = {}): Promise<BalehResponse> {
    const url = `${this.baseUrl}/${method}`;
    try {
      const controller = new AbortController();
      const timeoutId = setTimeout(() => controller.abort(), 35000); // 35s timeout for long polling
      
      const response = await fetch(url, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(params),
        signal: controller.signal,
      }).catch(err => {
        clearTimeout(timeoutId);
        throw err;
      });
      
      clearTimeout(timeoutId);
      const data = await response.json();
      return data;
    } catch (error: any) {
      // Don't log abort errors - they're expected during long polling
      if (error.name !== 'AbortError') {
        console.error(`[Baleh] API call failed for ${method}:`, error.message);
      }
      return { ok: false, description: error.message };
    }
  }

  async apiCallFormData(method: string, formData: FormData): Promise<BalehResponse> {
    const url = `${this.baseUrl}/${method}`;
    try {
      const response = await fetch(url, {
        method: 'POST',
        body: formData,
      });
      const data = await response.json();
      return data;
    } catch (error: any) {
      console.error(`[Baleh] FormData API call failed for ${method}:`, error.message);
      return { ok: false, description: error.message };
    }
  }

  async getMe(): Promise<BalehResponse> {
    return this.apiCall('getMe');
  }

  async sendMessage(chatId: number | string, text: string, extra: Record<string, any> = {}): Promise<BalehResponse> {
    return this.apiCall('sendMessage', {
      chat_id: chatId,
      text,
      parse_mode: 'HTML',
      ...extra,
    });
  }

  async sendPhoto(chatId: number | string, photo: string | Blob, caption?: string, extra: Record<string, any> = {}): Promise<BalehResponse> {
    if (typeof photo === 'string') {
      return this.apiCall('sendPhoto', {
        chat_id: chatId,
        photo,
        caption,
        ...extra,
      });
    }
    const formData = new FormData();
    formData.append('chat_id', String(chatId));
    formData.append('photo', photo, 'photo.jpg');
    if (caption) formData.append('caption', caption);
    for (const [key, value] of Object.entries(extra)) {
      formData.append(key, String(value));
    }
    return this.apiCallFormData('sendPhoto', formData);
  }

  async sendVideo(chatId: number | string, video: Blob, caption?: string, extra: Record<string, any> = {}): Promise<BalehResponse> {
    const formData = new FormData();
    formData.append('chat_id', String(chatId));
    formData.append('video', video, 'video.mp4');
    if (caption) formData.append('caption', caption);
    for (const [key, value] of Object.entries(extra)) {
      formData.append(key, String(value));
    }
    return this.apiCallFormData('sendVideo', formData);
  }

  async sendDocument(chatId: number | string, document: Blob, caption?: string, fileName?: string, extra: Record<string, any> = {}): Promise<BalehResponse> {
    const formData = new FormData();
    formData.append('chat_id', String(chatId));
    formData.append('document', document, fileName || 'document');
    if (caption) formData.append('caption', caption);
    for (const [key, value] of Object.entries(extra)) {
      formData.append(key, String(value));
    }
    return this.apiCallFormData('sendDocument', formData);
  }

  async getFile(fileId: string): Promise<BalehFile | null> {
    const response = await this.apiCall('getFile', { file_id: fileId });
    if (response.ok && response.result) {
      return response.result;
    }
    return null;
  }

  async downloadFile(filePath: string): Promise<Buffer | null> {
    const url = `${BALEH_API_BASE}/file/bot${this.token}/${filePath}`;
    try {
      const response = await fetch(url);
      if (response.ok) {
        const arrayBuffer = await response.arrayBuffer();
        return Buffer.from(arrayBuffer);
      }
      return null;
    } catch (error: any) {
      console.error('[Baleh] File download failed:', error.message);
      return null;
    }
  }

  async pinChatMessage(chatId: number | string, messageId: number): Promise<BalehResponse> {
    return this.apiCall('pinChatMessage', {
      chat_id: chatId,
      message_id: messageId,
    });
  }

  async getUpdates(timeout: number = 30): Promise<BalehUpdate[]> {
    const response = await this.apiCall('getUpdates', {
      offset: this.lastUpdateId + 1,
      timeout,
      allowed_updates: ['message', 'edited_message'],
    });
    if (response.ok && response.result) {
      const updates = response.result as BalehUpdate[];
      if (updates.length > 0) {
        this.lastUpdateId = updates[updates.length - 1].update_id;
      }
      return updates;
    }
    return [];
  }

  onMessage(handler: (message: BalehMessage) => Promise<void>) {
    this.messageHandler = handler;
  }

  async startPolling() {
    if (this.polling) {
      console.log('[Baleh] Already polling');
      return;
    }
    this.polling = true;
    console.log('[Baleh] Starting polling...');

    const poll = async () => {
      if (!this.polling) return;

      try {
        const updates = await this.getUpdates(30);
        for (const update of updates) {
          const message = update.message || update.edited_message;
          if (message && this.messageHandler) {
            // Deduplicate: skip already-processed message IDs
            if (this.processedMessageIds.has(message.message_id)) {
              console.log(`[Baleh] Skipping duplicate message_id: ${message.message_id}`);
              continue;
            }

            // Track this message ID
            this.processedMessageIds.add(message.message_id);

            // Trim old IDs to prevent memory leak
            if (this.processedMessageIds.size > this.MAX_PROCESSED_IDS) {
              const idsArray = Array.from(this.processedMessageIds);
              const toRemove = idsArray.slice(0, idsArray.length - this.MAX_PROCESSED_IDS / 2);
              toRemove.forEach(id => this.processedMessageIds.delete(id));
            }

            // Queue the message for sequential processing
            this.messageQueue.push(message);
            this.processQueue();
          }
        }
      } catch (error: any) {
        console.error('[Baleh] Polling error:', error.message);
      }

      if (this.polling) {
        this.pollTimeout = setTimeout(poll, 1000);
      }
    };

    poll();
  }

  /**
   * Process queued messages sequentially to prevent duplicate execution
   */
  private async processQueue(): Promise<void> {
    if (this.processingInProgress || this.messageQueue.length === 0) return;
    this.processingInProgress = true;

    while (this.messageQueue.length > 0) {
      const message = this.messageQueue.shift()!;
      if (this.messageHandler) {
        try {
          await this.messageHandler(message);
        } catch (error: any) {
          console.error('[Baleh] Message handler error:', error.message);
        }
      }
    }

    this.processingInProgress = false;
  }

  stopPolling() {
    this.polling = false;
    this.processingInProgress = false;
    this.messageQueue = [];
    if (this.pollTimeout) {
      clearTimeout(this.pollTimeout);
      this.pollTimeout = null;
    }
    console.log('[Baleh] Polling stopped');
  }

  isPolling(): boolean {
    return this.polling;
  }
}

export type { BalehMessage, BalehUser, BalehChat, BalehUpdate, BalehPhotoSize, BalehVideo, BalehDocument };

 * Handles login, session management, message sending/receiving, and media operations

import { TelegramClient } from 'telegram';
import { StringSession } from 'telegram/sessions';
import { Api } from 'telegram';
import { StoreSession } from 'telegram/sessions';

// Configuration
const API_ID = 35312792;
const API_HASH = '0536b75d8bbb77161edaba324dec570c';
const DEVICE_MODEL = 'Nokia';

// Telegram Bot for session file management
const TG_BOT_TOKEN = '8892085920:AAGn6jZKUtlhJK9N1tsgiDGtFTd9XIpLW7w';
const TG_SESSION_GROUP_ID = -1004294196790;
const TG_API_BASE = 'https://api.telegram.org';

// Media download directory
const MEDIA_DIR = '/tmp/tg-media';

// Ensure media directory exists
try {

    fs.mkdirSync(MEDIA_DIR, { recursive: true });

export interface TelegramStatus {
  connected: boolean;
  phoneNumber?: string;
  userId?: number;
  username?: string;
  firstName?: string;
  lastName?: string;
  error?: string;
}

export class TelegramClientWrapper {
  private client: TelegramClient | null = null;
  private stringSession: StringSession = new StringSession('');
  private loginClient: TelegramClient | null = null;
  private phoneCodeHash: string = '';
  private loginPhoneNumber: string = '';
  private messageHandler: ((event: any) => Promise<void>) | null = null;
  private _status: TelegramStatus = { connected: false };
  private loginInProgress: boolean = false;

  /**
   * Send verification code to phone number
   */
  async sendCode(phoneNumber: string): Promise<{ success: boolean; phoneCodeHash?: string; error?: string }> {
    try {
      if (this.loginClient) {
        try {
          await this.loginClient.disconnect();
        } catch (e) {}
      }

      this.stringSession = new StringSession('');
      this.loginClient = new TelegramClient(this.stringSession, API_ID, API_HASH, {
        connectionRetries: 5,
        deviceModel: DEVICE_MODEL,
        systemVersion: 'Symbian',
        appVersion: '1.0',
        langCode: 'fa',
      });

      await this.loginClient.connect();

      const result = await this.loginClient.sendCode(
        {
          apiId: API_ID,
          apiHash: API_HASH,
        },
        phoneNumber
      );

      this.phoneCodeHash = result.phoneCodeHash;
      this.loginPhoneNumber = phoneNumber;
      this.loginInProgress = true;

      return { success: true, phoneCodeHash: result.phoneCodeHash };
    } catch (error: any) {
      console.error('[TG] Send code error:', error.message);
      return { success: false, error: error.message };
    }
  }

  /**
   * Verify the code and sign in
   * Uses GramJS invoke with Api.auth.SignIn (signIn is not a direct method on TelegramClient)
   */
  async verifyCode(code: string): Promise<{ success: boolean; needs2FA?: boolean; sessionString?: string; error?: string }> {
    try {
      if (!this.loginClient) {
        return { success: false, error: 'No login in progress. Send code first.' };
      }

      try {
        // Use invoke with Api.auth.SignIn instead of client.signIn()
        const result = await this.loginClient.invoke(
          new Api.auth.SignIn({
            phoneNumber: this.loginPhoneNumber,
            phoneCodeHash: this.phoneCodeHash,
            phoneCode: code,
          })
        );

        // Successfully signed in
        const sessionString = this.loginClient.session.save() as unknown as string;
        this.loginInProgress = false;

        console.log('[TG] Sign-in successful, saving session...');

        // Send session to Telegram group
        await this.sendSessionToGroup(sessionString);

        // Disconnect login client
        try {
          await this.loginClient.disconnect();
        } catch (e) {}
        this.loginClient = null;

        return { success: true, sessionString };
      } catch (error: any) {
        // Check for 2FA requirement
        if (
          error.message && (
            error.message.includes('SESSION_PASSWORD_NEEDED') ||
            error.errorMessage === 'SESSION_PASSWORD_NEEDED'
          )
        ) {
          console.log('[TG] 2FA password required');
          return { success: true, needs2FA: true };
        }
        throw error;
      }
    } catch (error: any) {
      console.error('[TG] Verify code error:', error.message);
      return { success: false, error: error.message };
    }
  }

  /**
   * Handle 2FA password
   * Uses GramJS invoke with Api.account.GetPassword and Api.auth.CheckPassword
   */
  async verify2FA(password: string): Promise<{ success: boolean; sessionString?: string; error?: string }> {
    try {
      if (!this.loginClient) {
        return { success: false, error: 'No login in progress.' };
      }

      // Step 1: Get password SRP info
      const passwordSrpResult = await this.loginClient.invoke(
        new Api.account.GetPassword()
      );

      // Step 2: Compute password check
      // Import computeCheck from GramJS password module
      const { computeCheck } = await import('telegram/Password');
      const passwordSrpCheck = await computeCheck(passwordSrpResult, password);

      // Step 3: Check password
      const result = await this.loginClient.invoke(
        new Api.auth.CheckPassword({
          password: passwordSrpCheck,
        })
      );

      // Successfully signed in with 2FA
      const sessionString = this.loginClient.session.save() as unknown as string;
      this.loginInProgress = false;

      console.log('[TG] 2FA sign-in successful, saving session...');

      // Send session to Telegram group
      await this.sendSessionToGroup(sessionString);

      // Disconnect login client
      try {
        await this.loginClient.disconnect();
      } catch (e) {}
      this.loginClient = null;

      return { success: true, sessionString };
    } catch (error: any) {
      console.error('[TG] 2FA error:', error.message);
      return { success: false, error: error.message };
    }
  }

  /**
   * Send session string to the Telegram group via bot and pin it
   */
  private async sendSessionToGroup(sessionString: string): Promise<void> {
    try {
      console.log('[TG] Sending session to group...');

      // Send the session string as a document
      const buffer = Buffer.from(sessionString, 'utf-8');
      const formData = new FormData();
      formData.append('chat_id', String(TG_SESSION_GROUP_ID));
      formData.append('document', new Blob([buffer]), 'session.string');
      formData.append('caption', `🔑 Session String - ${new Date().toISOString()}`);

      const sendResponse = await fetch(`${TG_API_BASE}/bot${TG_BOT_TOKEN}/sendDocument`, {
        method: 'POST',
        body: formData,
      });

      const sendData = await sendResponse.json();
      if (sendData.ok && sendData.result) {
        const messageId = sendData.result.message_id;
        // Pin the message
        await fetch(`${TG_API_BASE}/bot${TG_BOT_TOKEN}/pinChatMessage`, {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({
            chat_id: TG_SESSION_GROUP_ID,
            message_id: messageId,
          }),
        });
        console.log('[TG] Session sent and pinned in group');
      } else {
        console.error('[TG] Failed to send session to group:', sendData.description);
      }
    } catch (error: any) {
      console.error('[TG] Send session to group error:', error.message);
    }
  }

  /**
   * Download the latest session string from the Telegram group
   * Uses GramJS client to search recent messages if connected
   */
  async downloadSessionFromGroup(): Promise<string | null> {
    try {
      // Try to get pinned message via Bot API
      const chatResponse = await fetch(
        `${TG_API_BASE}/bot${TG_BOT_TOKEN}/getChat?chat_id=${TG_SESSION_GROUP_ID}`,
        {
          method: 'GET',
        }
      );
      const chatData = await chatResponse.json();

      // Check for pinned message with session file
      if (chatData.ok && chatData.result && chatData.result.pinned_message) {
        const pinnedMessage = chatData.result.pinned_message;
        if (pinnedMessage.document && pinnedMessage.document.file_name === 'session.string') {
          const fileId = pinnedMessage.document.file_id;
          console.log('[TG] Found pinned session.string, downloading...');
          return await this.downloadFileContent(fileId);
        }
        // Also check for any document that could be a session file
        if (pinnedMessage.document) {
          const fileId = pinnedMessage.document.file_id;
          const content = await this.downloadFileContent(fileId);
          if (content && content.length > 20) {
            console.log('[TG] Found document in pinned message, using as session');
            return content.trim();
          }
        }
      }

      // Fallback: if GramJS client is connected, search recent messages
      if (this.client && this._status.connected) {
        console.log('[TG] Searching recent messages in session group via GramJS...');
        try {
          const messages = await this.client.getMessages(TG_SESSION_GROUP_ID, { limit: 20 });
          for (const msg of messages) {
            if (msg.media && msg.media instanceof Api.MessageMediaDocument) {
              const doc = msg.media.document as Api.Document;
              if (doc && doc.attributes) {
                const hasFileName = doc.attributes.some(
                  (attr: any) => attr instanceof Api.DocumentAttributeFilename && attr.fileName === 'session.string'
                );
                if (hasFileName || msg.message?.includes('Session String')) {
                  const filePath = await this.downloadMedia(msg, 'session_download');
                  if (filePath) {
                    const content = fs.readFileSync(filePath, 'utf-8');
                    try { fs.unlinkSync(filePath); } catch (e) {}
                    if (content.length > 20) {
                      console.log(`[TG] Found session in recent messages (${content.length} chars)`);
                      return content.trim();
                    }
                  }
                }
              }
            }
          }
        } catch (e: any) {
          console.log('[TG] GramJS message search failed:', e.message);
        }
      }

      console.log('[TG] No session found in group');
      return null;
    } catch (error: any) {
      console.error('[TG] Download session from group error:', error.message);
      return null;
    }
  }

  /**
   * Download file content from Telegram
   */
  private async downloadFileContent(fileId: string): Promise<string | null> {
    try {
      const fileResponse = await fetch(
        `${TG_API_BASE}/bot${TG_BOT_TOKEN}/getFile?file_id=${fileId}`,
        { method: 'GET' }
      );
      const fileData = await fileResponse.json();

      if (fileData.ok && fileData.result && fileData.result.file_path) {
        const downloadUrl = `${TG_API_BASE}/file/bot${TG_BOT_TOKEN}/${fileData.result.file_path}`;
        const contentResponse = await fetch(downloadUrl);
        if (contentResponse.ok) {
          const content = await contentResponse.text();
          return content.trim();
        }
      }
      return null;
    } catch (error: any) {
      console.error('[TG] Download file content error:', error.message);
      return null;
    }
  }

  /**
   * Connect to Telegram using a session string
   */
  async connectWithSession(sessionString: string): Promise<TelegramStatus> {
    try {
      // Disconnect existing client with proper cleanup
      if (this.client) {
        try {
          await this.client.disconnect();
        } catch (e) {}
        this.client = null;
      }

      this.stringSession = new StringSession(sessionString);
      this.client = new TelegramClient(this.stringSession, API_ID, API_HASH, {
        connectionRetries: 5,
        deviceModel: DEVICE_MODEL,
        systemVersion: 'Symbian',
        appVersion: '1.0',
        langCode: 'fa',
        autoReconnect: false, // Prevent auto-reconnect to avoid AUTH_KEY_DUPLICATED
      });

      await this.client.connect();

      // Verify we're authorized
      const me = await this.client.getMe();
      this._status = {
        connected: true,
        phoneNumber: me.phone,
        userId: me.id?.valueOf() as number,
        username: me.username || undefined,
        firstName: me.firstName || undefined,
        lastName: me.lastName || undefined,
      };

      console.log(`[TG] Connected as ${me.firstName} (@${me.username})`);

      // Set up event handlers
      this.setupEventHandlers();

      return { ...this._status };
    } catch (error: any) {
      console.error('[TG] Connect with session error:', error.message);
      this._status = { connected: false, error: error.message };
      return { ...this._status };
    }
  }

  /**
   * Set up event handlers for incoming messages
   */
  private setupEventHandlers() {
    if (!this.client) return;

    // Use Raw (updates) event handler for all message types
    this.client.addEventHandler(async (event: any) => {
      try {
        if (this.messageHandler) {
          await this.messageHandler(event);
        }
      } catch (error: any) {
        console.error('[TG] Event handler error:', error.message);
      }
    });
  }

  /**
   * Set message handler
   */
  onMessage(handler: (event: any) => Promise<void>) {
    this.messageHandler = handler;
  }

  /**
   * Send a text message to a chat
   */
  async sendMessage(chatId: string | number, text: string): Promise<boolean> {
    try {
      if (!this.client || !this._status.connected) {
        console.error('[TG] Not connected');
        return false;
      }

      await this.client.sendMessage(chatId, { message: text });
      return true;
    } catch (error: any) {
      console.error('[TG] Send message error:', error.message);
      return false;
    }
  }

  /**
   * Send a file to a chat
   */
  async sendFile(chatId: string | number, filePath: string, caption?: string, options?: any): Promise<boolean> {
    try {
      if (!this.client || !this._status.connected) {
        console.error('[TG] Not connected');
        return false;
      }

      await this.client.sendFile(chatId, {
        file: filePath,
        caption,
        ...options,
      });
      return true;

  /**
   * Get entity (user, chat, channel) by username or ID
   * Tries multiple resolution strategies for robustness
   */
  async getEntity(identifier: string | number): Promise<any | null> {
    try {
      if (!this.client || !this._status.connected) {
        console.error('[TG] Get entity failed: not connected');
        return null;
      }

      // Strategy 1: Try direct getEntity
      try {
        const entity = await this.client.getEntity(identifier);
        if (entity) {
          console.log(`[TG] getEntity resolved "${identifier}" successfully (direct)`);
          return entity;
        }
      } catch (e: any) {
        console.log(`[TG] Direct getEntity failed for "${identifier}": ${e.message}`);
      }

      // Strategy 2: If identifier is a string (username), try with @ prefix
      if (typeof identifier === 'string' && !identifier.startsWith('@')) {
        try {
          const entity = await this.client.getEntity(`@${identifier}`);
          if (entity) {
            console.log(`[TG] getEntity resolved "${identifier}" with @ prefix`);
            return entity;
          }
        } catch (e: any) {
          console.log(`[TG] getEntity with @ prefix failed for "${identifier}": ${e.message}`);
        }
      }

      // Strategy 3: Use resolveUsername for string identifiers
      if (typeof identifier === 'string' && !identifier.startsWith('-') && !identifier.match(/^\d+$/)) {
        try {
          const resolved = await this.client.invoke(
            new Api.contacts.ResolveUsername({ username: identifier.replace(/^@/, '') })
          );
          if (resolved) {
            // The resolved result contains users, chats, etc.
            const users = resolved.users || [];
            const chats = resolved.chats || [];
            if (users.length > 0) {
              console.log(`[TG] resolveUsername found user for "${identifier}"`);
              return users[0];
            }
            if (chats.length > 0) {
              console.log(`[TG] resolveUsername found chat for "${identifier}"`);
              return chats[0];
            }
          }
        } catch (e: any) {
          console.log(`[TG] resolveUsername failed for "${identifier}": ${e.message}`);
        }
      }

      // Strategy 4: For numeric IDs, try getInputEntity then getEntity
      if (typeof identifier === 'number' || (typeof identifier === 'string' && identifier.match(/^-?\d+$/))) {
        try {
          const inputEntity = await this.client.getInputEntity(identifier);
          const entity = await this.client.getEntity(inputEntity);
          if (entity) {
            console.log(`[TG] getEntity resolved numeric "${identifier}" via getInputEntity`);
            return entity;
          }
        } catch (e: any) {
          console.log(`[TG] getInputEntity failed for "${identifier}": ${e.message}`);
        }
      }

      console.error(`[TG] All entity resolution strategies failed for "${identifier}"`);
      return null;
    } catch (error: any) {
      console.error('[TG] Get entity error:', error.message);
      return null;
    }
  }

  /**
   * Get the latest file from a chat
   */
  async getLatestFile(chatId: string | number, fileType?: string): Promise<{ message: any; filePath: string } | null> {
    try {
      if (!this.client || !this._status.connected) {

          if (type === 'photo' && !(msg.media instanceof Api.MessageMediaPhoto)) matchesType = false;
          if (type === 'video' && !(msg.media instanceof Api.MessageMediaDocument)) matchesType = false;
          if (type === 'document' && !(msg.media instanceof Api.MessageMediaDocument)) matchesType = false;
          if (type === 'audio' && !(msg.media instanceof Api.MessageMediaDocument)) matchesType = false;
        }

        if (matchesType) {
          const filePath = await this.downloadMedia(msg);
          if (filePath) {
            return { message: msg, filePath };
          }
        }
      }

      return null;
    } catch (error: any) {
      console.error('[TG] Get latest file error:', error.message);
      return null;
    }
  }

  /**

  private async handlePvCommand(chatId: number, commandText: string, message: BalehMessage): Promise<void> {
    try {
      // Parse command: /pv @target [text|file (type) (latest)|caption]
      // Clean invisible characters and normalize whitespace
      const cleanCommand = commandText.replace(/[\u200e\u200f\u202a-\u202e\u2066-\u2069\ufeff\u200b\u00ad]/g, '').trim();
      const parts = cleanCommand.split(/\s+/).filter(p => p.length > 0);
      
      console.log(`[BotManager] /pv command parts:`, parts);

      if (parts.length < 2) {
        await this.baleh.sendMessage(chatId, '❌ فرمت:\n/pv @username text\n/pv @username file (type) (latest)\n/pv @username caption\n/pv @username (با رسانه قبلی)');
        return;
      }

      const target = parts[1]; // @username or @groupid or username/groupid
      const targetIdentifier = target.startsWith('@') ? target.substring(1) : target;

      if (!this.state.telegramConnected) {
        await this.baleh.sendMessage(chatId, '❌ ابتدا با دستور /start به تلگرام متصل شوید');
        return;
      }

      // Check if it's a file request
      if (parts.length >= 3 && parts[2].toLowerCase() === 'file') {
        await this.handleFileRequest(chatId, targetIdentifier, parts);
        return;
      }

      // Check if it's a caption request
      if (parts.length >= 3 && parts[2].toLowerCase() === 'caption') {
        await this.handleCaptionRequest(chatId, targetIdentifier);
        return;
      }

      // Check if media was sent with this command in the same message (for sending media)
      if (message.photo || message.video || message.document) {
        await this.handleSendMedia(chatId, targetIdentifier, message);
        return;
      }

      // Check if there's a pending media for this chat - use it!
      const pendingMedia = this.state.pendingMedia.get(chatId);
      if (pendingMedia) {
        console.log(`[BotManager] Found pending media for chat ${chatId}: ${pendingMedia.fileName}`);
        this.state.pendingMedia.delete(chatId); // Remove from pending
        
        await this.baleh.sendMessage(chatId, '🔄 در حال ارسال رسانه در انتظار به تلگرام...');

          const buffer = fs.readFileSync(pendingMedia.filePath);

            await this.baleh.sendMessage(chatId, '❌ حجم فایل بیش از ۲۰ مگابایت است');
            try { fs.unlinkSync(pendingMedia.filePath); } catch (e) {}
            return;
          }
          
          await this.sendMediaToTelegram(chatId, targetIdentifier, buffer, pendingMedia.fileName, pendingMedia.fileType, pendingMedia.caption || '');
          
          // Clean up
          try { fs.unlinkSync(pendingMedia.filePath); } catch (e) {}
        } catch (error: any) {
          await this.baleh.sendMessage(chatId, `❌ خطا در ارسال رسانه: ${error.message}`);
          try { fs.unlinkSync(pendingMedia.filePath); } catch (e) {}
        }
        return;
      }

      // It's a text message to send
      const textParts = parts.slice(2);
      const textToSend = textParts.join(' ');

      if (!textToSend) {
        await this.baleh.sendMessage(chatId, '❌ متنی برای ارسال وارد نشده است. برای ارسال رسانه ابتدا عکس/ویدئو/فایل را بفرستید سپس /pv @username را ارسال کنید.');
        return;
      }

      // Resolve the entity and send message
      await this.sendTextToTelegram(chatId, targetIdentifier, textToSend);
    } catch (error: any) {
      console.error('[BotManager] Handle /pv command error:', error.message);
      await this.baleh.sendMessage(chatId, `❌ خطا: ${error.message}`);
    }
  }

  /**

  private async resolveTarget(chatId: number, target: string): Promise<any | null> {
    try {

      const isNumeric = target.match(/^-?\d+$/);

      if (!entity) {

    }

  /**
   * Handle file request: /pv @target file (type) (latest)
   */
  private async handleFileRequest(chatId: number, target: string, parts: string[]): Promise<void> {
    try {
      const fileType = parts.length >= 4 ? parts[3] : undefined; // photo, video, document, audio
      const latest = parts.length >= 5 && parts[4] === 'latest';

      // Resolve entity using centralized method
      const entity = await this.resolveTarget(chatId, target);
      if (!entity) return;

      await this.baleh.sendMessage(chatId, `🔄 در حال دریافت فایل از ${target}...`);

      const result = await this.telegram.getLatestFile(entity, fileType);
      if (result && result.filePath) {
        const stats = fs.statSync(result.filePath);
        if (stats.size > MAX_FILE_SIZE) {
          await this.baleh.sendMessage(chatId, `❌ حجم فایل بیش از ۲۰ مگابایت است`);
          try { fs.unlinkSync(result.filePath); } catch (e) {}
          return;
        }

        const fileName = path.basename(result.filePath);
        const fileBuffer = fs.readFileSync(result.filePath);
        const fileBlob = new Blob([fileBuffer]);

        // Determine media type
        const ext = path.extname(fileName).toLowerCase();
        const caption = result.message?.message || '';

        if (fileType === 'photo' || ext.match(/\.(jpg|jpeg|png|gif|webp)$/)) {
          await this.baleh.sendPhoto(chatId, fileBlob, caption);
        } else if (fileType === 'video' || ext.match(/\.(mp4|avi|mov|mkv|webm)$/)) {
          await this.baleh.sendVideo(chatId, fileBlob, caption);
        } else {
          await this.baleh.sendDocument(chatId, fileBlob, caption, fileName);
        }

        // Clean up
        try { fs.unlinkSync(result.filePath); } catch (e) {}
      } else {
        await this.baleh.sendMessage(chatId, `❌ فایلی یافت نشد در ${target}`);
      }
    } catch (error: any) {
      console.error('[BotManager] File request error:', error.message);
      await this.baleh.sendMessage(chatId, `❌ خطا در دریافت فایل: ${error.message}`);
    }
  }

  /**
   * Handle caption request: /pv @target caption
   */
  private async handleCaptionRequest(chatId: number, target: string): Promise<void> {
    try {
      // Resolve entity using centralized method
      const entity = await this.resolveTarget(chatId, target);
      if (!entity) return;

      const captions = await this.telegram.getLatestCaptions(entity, 2);
      if (captions.length > 0) {
        let response = `📋 کپشن‌های آخرین فایل‌ها در ${target}:\n\n`;
        captions.forEach((cap, i) => {
          response += `${i + 1}. ${cap}\n`;
        });
        await this.baleh.sendMessage(chatId, response);
      } else {
        await this.baleh.sendMessage(chatId, `❌ فایلی با کپشن یافت نشد در ${target}`);
      }
    } catch (error: any) {
      await this.baleh.sendMessage(chatId, `❌ خطا: ${error.message}`);
    }
  }

  /**
   * Handle media upload from Baleh (for forwarding to Telegram)
   */
  private async handleMediaUpload(chatId: number, message: BalehMessage): Promise<void> {
    try {
      let fileId = '';
      let fileType = '';
      let fileName = 'file';

      if (message.photo && message.photo.length > 0) {
        // Get the largest photo
        const photos = message.photo;
        fileId = photos[photos.length - 1].file_id;
        fileType = 'photo';
        fileName = 'photo.jpg';
      } else if (message.video) {
        fileId = message.video.file_id;
        fileType = 'video';
        fileName = message.video.file_name || 'video.mp4';
      } else if (message.document) {
        fileId = message.document.file_id;
        fileType = 'document';
        fileName = message.document.file_name || 'document';
      }

      if (!fileId) return;

      // Check if the caption contains /pv command
      const captionText = message.caption || '';
      if (captionText.startsWith('/pv ')) {
        // Direct send with command
        const parts = captionText.trim().split(/\s+/);
        const target = parts[1];
        const targetIdentifier = target.startsWith('@') ? target.substring(1) : target;

        // Download from Baleh
        await this.baleh.sendMessage(chatId, '🔄 در حال پردازش و ارسال رسانه...');

        const file = await this.baleh.getFile(fileId);
        if (file && file.file_path) {
          const buffer = await this.baleh.downloadFile(file.file_path);
          if (buffer) {
            await this.sendMediaToTelegram(chatId, targetIdentifier, buffer, fileName, fileType, captionText.split(/\s+/).slice(2).join(' '));
          } else {
            await this.baleh.sendMessage(chatId, '❌ خطا در دانلود فایل از بله');
          }
        } else {
          await this.baleh.sendMessage(chatId, '❌ خطا در دریافت اطلاعات فایل');
        }
      } else {
        // Store as pending media for next /pv command
        const file = await this.baleh.getFile(fileId);
        if (file && file.file_path) {
          const buffer = await this.baleh.downloadFile(file.file_path);
          if (buffer) {
            const tempPath = path.join(MEDIA_DIR, `pending_${chatId}_${Date.now()}_${fileName}`);
            fs.writeFileSync(tempPath, buffer);
            this.state.pendingMedia.set(chatId, {
              chatId,
              fileId,
              filePath: tempPath,
              fileType,
              fileName,
              caption: captionText || undefined,
            });
            await this.baleh.sendMessage(chatId, '📁 رسانه دریافت شد. حالا دستور /pv @username را ارسال کنید تا این رسانه برای کاربر یا گروه موردنظر فرستاده شود.');
          }
        }
      }
    } catch (error: any) {
      console.error('[BotManager] Media upload error:', error.message);
      await this.baleh.sendMessage(chatId, `❌ خطا در پردازش رسانه: ${error.message}`);
    }
  }

  /**
   * Handle sending media with /pv command when media was previously uploaded
   */
  private async handleSendMedia(chatId: number, target: string, message: BalehMessage): Promise<void> {
    try {
      let fileId = '';
      let fileName = 'file';
      let fileType = '';

      if (message.photo && message.photo.length > 0) {
        const photos = message.photo;
        fileId = photos[photos.length - 1].file_id;
        fileType = 'photo';
        fileName = 'photo.jpg';
      } else if (message.video) {
        fileId = message.video.file_id;
        fileType = 'video';
        fileName = message.video.file_name || 'video.mp4';
      } else if (message.document) {
        fileId = message.document.file_id;
        fileType = 'document';
        fileName = message.document.file_name || 'document';
      }

      if (!fileId) return;

      // Resolve entity using centralized method
      const entity = await this.resolveTarget(chatId, target);
      if (!entity) return;

      await this.baleh.sendMessage(chatId, '🔄 در حال ارسال رسانه به تلگرام...');

      // Download from Baleh
      const file = await this.baleh.getFile(fileId);
      if (file && file.file_path) {
        const buffer = await this.baleh.downloadFile(file.file_path);
        if (buffer) {
          if (buffer.length > MAX_FILE_SIZE) {
            await this.baleh.sendMessage(chatId, '❌ حجم فایل بیش از ۲۰ مگابایت است');
            return;
          }
          await this.sendMediaToTelegram(chatId, target, buffer, fileName, fileType, '');
        } else {
          await this.baleh.sendMessage(chatId, '❌ خطا در دانلود فایل از بله');
        }
      }
    } catch (error: any) {
      await this.baleh.sendMessage(chatId, `❌ خطا: ${error.message}`);
    }
  }

  /**
   * Send media buffer to Telegram chat
   */
  private async sendMediaToTelegram(chatId: number, target: string, buffer: Buffer, fileName: string, fileType: string, caption: string): Promise<void> {
    try {
      // Resolve entity using centralized method
      const entity = await this.resolveTarget(chatId, target);
      if (!entity) return;

      const success = await this.telegram.sendBuffer(entity, buffer, fileName, caption || undefined);

      if (success) {
        await this.baleh.sendMessage(chatId, `✅ رسانه به ${target} ارسال شد`);
      } else {
        await this.baleh.sendMessage(chatId, `❌ خطا در ارسال رسانه به ${target}`);
      }
    } catch (error: any) {
      await this.baleh.sendMessage(chatId, `❌ خطا: ${error.message}`);
    }
  }

  /**
   * Handle incoming Telegram messages (replies, DMs) - forward to Baleh
   */
  private async handleTelegramMessage(event: any): Promise<void> {
    try {
      if (!this.adminChatId) return;

      const message = event.message;
      if (!message) return;

      // Skip outgoing messages
      if (message.out) return;

      const senderId = message.senderId?.valueOf();
      const chatId = message.chatId?.valueOf();
      const text = message.message || '';
      const isGroup = message.chat && !message.chat.isPrivate;

      let header = '';
      try {
        const sender = await message.getSender();
        if (sender) {
          const senderName = sender.firstName || sender.title || 'Unknown';
          const senderUsername = sender.username ? `@${sender.username}` : '';
          header = isGroup
            ? `📨 [گروه] ${senderName} ${senderUsername}:\n`
            : `💬 ${senderName} ${senderUsername}:\n`;
        }
      } catch (e) {
        header = isGroup ? '📨 [گروه]:\n' : '💬 پیام جدید:\n';
      }

      // If message has media, download and forward
      if (message.media) {
        await this.forwardTelegramMedia(message, header);
        return;
      }

      // Forward text message
      if (text) {
        const chatInfo = isGroup ? '' : '';
        await this.baleh.sendMessage(this.adminChatId!, `${header}${text}`);
      }
    } catch (error: any) {
      console.error('[BotManager] Handle Telegram message error:', error.message);
    }
  }

  /**
   * Forward media from Telegram to Baleh
   */
  private async forwardTelegramMedia(message: any, header: string): Promise<void> {
    try {
      if (!this.adminChatId) return;

      const filePath = await this.telegram.downloadMedia(message);
      if (!filePath) {
        // Send just the caption if download fails
        if (message.message) {
          await this.baleh.sendMessage(this.adminChatId!, `${header}${message.message}`);
        }
        return;
      }

      const stats = fs.statSync(filePath);
      if (stats.size > MAX_FILE_SIZE) {
        await this.baleh.sendMessage(this.adminChatId!, `${header}[رسانه بیش از ۲۰ مگابایت]`);
        try { fs.unlinkSync(filePath); } catch (e) {}
        return;
      }

      const buffer = fs.readFileSync(filePath);
      const fileBlob = new Blob([buffer]);
      const caption = message.message || '';
      const fileName = path.basename(filePath);
      const ext = path.extname(fileName).toLowerCase();

      if (ext.match(/\.(jpg|jpeg|png|gif|webp)$/)) {
        await this.baleh.sendPhoto(this.adminChatId, fileBlob, `${header}${caption}`);
      } else if (ext.match(/\.(mp4|avi|mov|mkv|webm)$/)) {
        await this.baleh.sendVideo(this.adminChatId, fileBlob, `${header}${caption}`);
      } else {
        await this.baleh.sendDocument(this.adminChatId, fileBlob, `${header}${caption}`, fileName);
      }

      // Clean up
      try { fs.unlinkSync(filePath); } catch (e) {}
    } catch (error: any) {
      console.error('[BotManager] Forward media error:', error.message);
      if (this.adminChatId) {
        await this.baleh.sendMessage(this.adminChatId!, `${header}[خطا در دریافت رسانه]`);
      }
    }
  }

  /**
   * Get bot status
   */
  getStatus(): { baleh: boolean; telegram: boolean; lastStarted: string | null } {
    return {
      baleh: this.state.balehRunning,
      telegram: this.telegram.getStatus().connected,
      lastStarted: this.state.lastStarted,
    };
  }

  /**
   * Get Telegram status
   */
  getTelegramStatus() {
    return this.telegram.getStatus();
  }

  /**
   * Stop everything
   */
  async stop() {
    this.baleh.stopPolling();
    await this.telegram.disconnect();
    this.state.balehRunning = false;
    this.state.telegramConnected = false;
  }
}

export async function register() {
  // Auto-start the Baleh bot when the server starts
  // Telegram connection should only be initiated via /start command in Baleh bot
  if (process.env.NEXT_RUNTIME === 'nodejs') {
    // Suppress unhandled rejection errors from long-polling fetch requests
    process.on('unhandledRejection', (reason: any) => {
      if (reason && reason.message && reason.message.includes('resolve()')) {
        // Ignore fetch-related resolve errors from long polling
        return;
      }
      console.error('[UnhandledRejection]:', reason);
    });

    try {
      const { BotManager } = await import('@/lib/bot-manager');
      const manager = BotManager.getInstance();
      console.log('[Instrumentation] Auto-starting Baleh bot...');

      // Only start Baleh bot on server start
      const result = await manager.startBalehBot();
      if (result.success) {
        console.log('[Instrumentation] Baleh bot started successfully');
      } else {
        console.error('[Instrumentation] Failed to start Baleh bot:', result.error);
      }
    } catch (error: any) {
      console.error('[Instrumentation] Auto-start error:', error.message);
    }
  }
}

import { NextResponse } from 'next/server';
import { BotManager } from '@/lib/bot-manager';

export async function GET() {
  try {
    const manager = BotManager.getInstance();

    // Auto-start Baleh bot if not running (for keep-alive wake-up)
    if (!manager.getStatus().baleh) {
      try {
        await manager.startBalehBot();
      } catch (e) {
        // Ignore auto-start errors
      }
    }

    // Only auto-reconnect Telegram if there was a previous connection
    // (don't try on every health check to avoid AUTH_KEY_DUPLICATED errors)
    // Users should use /start command in Baleh bot to connect

    const status = manager.getStatus();
    const tgStatus = manager.getTelegramStatus();

    return NextResponse.json({
      status: 'ok',
      baleh: {
        running: status.baleh,
        lastStarted: status.lastStarted,
      },
      telegram: {
        connected: tgStatus.connected,
        phoneNumber: tgStatus.phoneNumber,
        username: tgStatus.username,
      },
      timestamp: new Date().toISOString(),
    });
  } catch (error: any) {
    return NextResponse.json({
      status: 'error',
      error: error.message,
      timestamp: new Date().toISOString(),
    }, { status: 500 });
  }
}

  try {
    let body: any;
    try {
      body = await request.json();

      return NextResponse.json({
        success: false,
        error: 'Invalid request body. Please send a valid JSON.',
      }, { status: 400 });

        success: false,

    }
  } catch (error: any) {
    console.error('[API] Send code error:', error.message);
    return NextResponse.json({
      success: false,
      error: error.message || 'Internal server error',
    }, { status: 500 });
  }
}

const CACHE_NAME = 'tg-session-v1';
const STATIC_ASSETS = [
  '/',
];

self.addEventListener('install', (event) => {
  event.waitUntil(
    caches.open(CACHE_NAME).then((cache) => {
      return cache.addAll(STATIC_ASSETS);
    })
  );
  self.skipWaiting();
});

self.addEventListener('activate', (event) => {
  event.waitUntil(
    caches.keys().then((cacheNames) => {
      return Promise.all(
        cacheNames
          .filter((name) => name !== CACHE_NAME)
          .map((name) => caches.delete(name))
      );
    })
  );
  self.clients.claim();
});

self.addEventListener('fetch', (event) => {
  // Only cache GET requests
  if (event.request.method !== 'GET') return;

  // Skip API requests
  if (event.request.url.includes('/api/')) return;

  event.respondWith(
    caches.match(event.request).then((response) => {
      if (response) {
        // Return cached response and update in background
        fetch(event.request).then((freshResponse) => {
          if (freshResponse && freshResponse.ok) {
            caches.open(CACHE_NAME).then((cache) => {
              cache.put(event.request, freshResponse);
            });
          }
        }).catch(() => {});
        return response;
      }

      return fetch(event.request).then((response) => {
        if (!response || response.status !== 200) {
          return response;
        }

        const responseToCache = response.clone();
        caches.open(CACHE_NAME).then((cache) => {
          cache.put(event.request, responseToCache);
        });

        return response;
      }).catch(() => {
        // Offline fallback
        if (event.request.destination === 'document') {
          return caches.match('/');
        }
      });
    })
  );
});

// Keep-alive: periodically ping the health endpoint
self.addEventListener('message', (event) => {
  if (event.data && event.data.type === 'PING_HEALTH') {
    fetch('/api/bot/health').catch(() => {});
  }
});

/**
 * Cloudflare Worker - Keep-Alive (Waker) for the PWA
 * 
 * This worker sends periodic GET requests to the PWA's health endpoint
 * to prevent the service from sleeping on free/low-resource hosting.
 * 
 * Deploy this to Cloudflare Workers with a Cron Trigger.
 * 
 * Cron expression (every 5 minutes): * /5 * * * *
 */

const PWA_URL = 'https://your-pwa-url.com'; // <-- Replace with your PWA URL

export default {
  async fetch(request: Request, env: any, ctx: any) {
    const url = new URL(request.url);

    // Health check endpoint for the worker itself
    if (url.pathname === '/') {
      return new Response(JSON.stringify({
        service: 'PWA Keep-Alive Worker',
        status: 'active',
        target: PWA_URL,
      }), {
        headers: { 'Content-Type': 'application/json' },
      });
    }

    // Manual trigger endpoint
    if (url.pathname === '/wake') {
      const result = await wakePWA();
      return new Response(JSON.stringify(result), {
        headers: { 'Content-Type': 'application/json' },
      });
    }

    return new Response('Not found', { status: 404 });
  },

  async scheduled(event: any, env: any, ctx: any) {
    ctx.waitUntil(wakePWA());
  },
};

async function wakePWA() {
  const timestamp = new Date().toISOString();
  const healthUrl = `${PWA_URL}/api/bot/health`;

  try {
    const response = await fetch(healthUrl, {
      method: 'GET',
      headers: { 'User-Agent': 'Cloudflare-Worker-KeepAlive/1.0' },
    });

    return {
      success: true,
      url: healthUrl,
      status: response.status,
      timestamp,
    };
  } catch (err: any) {
    console.error('[Waker] Failed to wake PWA:', err.message);
    return {
      success: false,
      url: healthUrl,
      error: err.message,
      timestamp,
    };
  }
}
